    ```typescript
    import { useSortable } from '@dnd-kit/sortable';
    import { CSS } from '@dnd-kit/utilities';
    import { WarningBadge } from '../WarningBadge/WarningBadge';

    interface IssueCardProps {
      issue: BacklogIssue;
      laneId: string;          // NEW: 親LaneのID
      milestonePrefix: string; // NEW: プレフィックス（マルチMS判定用）
    }

    export function IssueCard({ issue, laneId, milestonePrefix }: IssueCardProps) {
      // マルチマイルストーン判定: プレフィックス一致のマイルストーンが2以上
      const prefixMilestones = issue.milestone.filter((m) => m.name.startsWith(milestonePrefix));
      const isMultiMilestone = prefixMilestones.length > 1;

      // 他のマイルストーン名（現在のレーンのマイルストーン以外、WarningBadgeのツールチップ用）
      // D-06: "他のマイルストーン: Sprint-2505, Release-v2.1"
      const otherMilestoneNames = issue.milestone
        .filter((m) => `milestone-${m.id}` !== laneId)
        .map((m) => m.name);

      const {
        attributes,
        listeners,
        setNodeRef,
        transform,
        transition,
        isDragging,
      } = useSortable({
        id: issue.id,
        disabled: isMultiMilestone, // D-08: multi-milestone cards cannot drag
      });

      const style = {
        transform: CSS.Transform.toString(transform),
        transition,
      };

      // 既存のhandleClick (Phase 4)
      const hostUrl = useSettingsStore((s) => s.settings.hostUrl);
      const handleClick = async () => {
        const host = hostUrl.replace(/^https?:\/\//, '');
        const url = `https://${host}/view/${issue.issueKey}`;
        try {
          await openUrl(url);
        } catch {
          // opener failure is non-critical
        }
      };

      return (
        <div
          ref={setNodeRef}
          style={style}
          {...attributes}
          {...listeners}
          className={`${styles.card} ${isDragging ? styles.cardDragging : ''} ${isMultiMilestone ? styles.cardDragDisabled : ''}`}
          onClick={handleClick}
          role="link"
          aria-label={`${issue.issueKey}をBacklogで開く`}
        >
          <div className={styles.line1}>
            <span className={styles.issueKey}>{issue.issueKey}</span>
            {isMultiMilestone && (
              <WarningBadge otherMilestones={otherMilestoneNames} />
            )}
            <StatusBadge name={issue.status.name} color={issue.status.color} />
          </div>
          <div className={styles.summary}>{issue.summary}</div>
          <div className={styles.line3}>
            {issue.assignee !== null ? (
              <span className={styles.assignee}>{issue.assignee.name}</span>
            ) : (
              <span className={styles.noAssignee}>---</span>
            )}
            <PriorityIndicator priority={issue.priority} />
          </div>
        </div>
      );
    }
    ```

    ```typescript
    // issueIdから所属レーンIDを逆引き
    function findLaneContaining(data: BoardData, itemId: number | string): string | null {
      const id = typeof itemId === 'string' ? Number(itemId) : itemId;
      if (data.unassignedIssues.some((i) => i.id === id)) return 'unassigned';
      for (const mwi of data.milestones) {
        if (mwi.issues.some((i) => i.id === id)) return `milestone-${mwi.milestone.id}`;
      }
      return null;
    }

    // over.id からレーンIDを取得（overがレーンの場合はそのまま、アイテムの場合はそのアイテムの所属レーン）
    function resolveOverLaneId(data: BoardData, overId: number | string): string | null {
      const overStr = String(overId);
      // レーンIDの場合
      if (overStr === 'unassigned' || overStr.startsWith('milestone-')) return overStr;
      // アイテムIDの場合
      return findLaneContaining(data, overId);
    }

    export function Board() {

      const handleDragStart = (event: DragStartEvent) => {
        if (!data) return;
        const issueId = event.active.id as number;
        // 全レーンからissueを検索

        setActiveIssue(found);
      };

        setActiveIssue(null);
        if (!data || !event.over) return;

        const fromLaneId = findLaneContaining(data, event.active.id as number);
        const toLaneId = resolveOverLaneId(data, event.over.id);

        if (fromLaneId && toLaneId && fromLaneId !== toLaneId) {
          moveIssue(event.active.id as number, fromLaneId, toLaneId);
        }
      };

      // 既存のステータス分岐（loading, error）は変更なし
      // ... useEffect, if blocks ...

      // loaded状態の場合、DndContextでラップ
      // return (
    ```
